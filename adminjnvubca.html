⚡ Supabase Setup

Step 1: Create free project at supabase.com
Step 2: Run this SQL in SQL Editor: 

create table leads (
  id uuid default gen_random_uuid() primary key,
  name text,
  mobile text,
  semester integer,
  subject text,
  note_title text,
  created_at timestamptz default now()
);

alter table leads enable row level security;

create policy "insert"
on leads
for insert
to anon
with check (true);

create policy "select"
on leads
for select
to authenticated
using (true);

create policy "delete"
on leads
for delete
to authenticated
using (true);

Step 3: Settings → API → copy Project URL + anon key
Step 4: Paste inside js/admin.js and js/notes.js
